********************************* 多数元素 ***********************************
投票选举中，要求某一人的得票数要超过所有人数的一半当选主席，现给你一个投票数组nums nums[i]表示第i位投票人的投票结果 ， 请你找出此次可能当选的主席
[多数元素](https://leetcode.cn/problems/majority-element/)
*      int candle = 0 , cnt = 0;
       for(int num : nums){
         if(cnt == 0){
           cnt++
           candle = num;
           continue;
           // 说明当前阶段一定没有谁的得票超过总人数的一半 一定不能当选 需要重新选择一个人
         }
         if(num == candle){
           cnt++;
         }else cnt--;
         // 进行检查 这个人的票数是否真的大于总人数的一半
         cnt = 0;
         for(int num : nums){
           if(num == candle) cnt++;
         }
         if(cnt > nums.length / 2) return candle;
         return -1;
